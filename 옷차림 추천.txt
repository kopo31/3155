#include <stdio.h>

void main()
{
    int a = 0;
    int b = 0;
    scanf("%d", &a);
    
    if(a >= 30) {
        b = 1;
    } else if(a >= 25 && a <= 30) {
        b = 2;
    } else if(a >= 15 && a <= 25) {
        b = 3;
    } else if(a >= 5 && a <= 15) {
        b = 4;
    } else if(a >= 0 && a <= 5) {
        b = 5;
    }

    switch(b) 
    {
        case 1:
            printf("30도 이상");
            printf("너무 더워요! 반팔과 반바지를 입으세요.");
            break;            
        case 2:
            printf("25도 이상 30도이하");
            printf("따뜻한 날씨입니다! 가벼운 옷을 입으세요.");
            break;                       
        case 3:
            printf("15도이상 25도 이하");
            printf("선선한 날씨입니다! 가벼운 겉옷을 챙기세요.");
            break;            
        case 4:
            printf("5도 이상 15도 이하");
            printf("조금 쌀쌀합니다! 따뜻한 옷을 입으세요요."); 
            break;             
        case 5:
            printf("0도 이상 5도이하");
            printf("매우 춥습니다! 두꺼운 외투를 꼭 입으세요요."); 
            break;           
        default:
            printf("0도 미만");
            printf("한파 경고! 패딩과 장갑,모자를 착용하세요.");            
    }
    return 0;
}
