int main()
{
   union num {
       int intVal;
       float floatVal;
       char charVal;
       
   };
   
   union num u;
   u.intVal = 100;
   u.floatVal = 2.2;
   u.charVal = 'A';
   
   printf("%d %d\n", u.intVal, &u.intVal);
   printf("%f %d\n", u.floatVal, &u.floatVal);
   printf("%c %d\n", u.charVal, &u.charVal);
}
