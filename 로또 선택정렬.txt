#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main()
{
 int temp = 0;
 int number[6];
 srand(time(NULL));
 
   for (int i = 0; i < 6; i++) //중복 여부 체크
   {
     number[i] = (int)(rand() % 45) + 1; //1~45 사이 숫자 생성
     for(int j = 0; j < i; j++)
     {
       if (number[i] == number[j]) //중복검사
       {
           i--; // 다시실행
       }
     }
     printf("%d ", number[i]);
     sleep(1);
   }
   for (int i = 0 ; i < 6; i++) 
   {  
     for (int j = i +1 ; j < 6 ; j++)
     {
       if (number[j] < number[i])
       {
         temp = number[i];
         number[i] = number[j];
         number[j] = temp;
       }
     }
   }

   for (int i = 0 ; i < 6; i++) 
   {
        printf("%d ", number[i]);  
   }
    return 0;
}
