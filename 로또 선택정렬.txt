#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main()
{
 int temp = 0;
 int number[6];
 srand(time(NULL));
 
   for (int i = 0; i < 6; i++)
   {
     number[i] = (int)(rand() % 45) + 1;
     for(int j = 0; j < i; j++)
     {
       if (number[i] == number[j])
       {
           i--;
       }
     }
     printf("%d ", number[i]);
     sleep(1);
   }
   for (int i = 0 ; i < 6; i++) 
   {  
     for (int j = i +1 ; j < 6 ; j++)
     {
       if (number[j] < number[i])
       {
         temp = number[i];
         number[i] = number[j];
         number[j] = temp;
       }
     }
   }

   for (int i = 0 ; i < 6; i++) 
   {
        printf("%d ", number[i]);  
   }
    return 0;
}
