#include <stdio.h>

struct student {
    char name[40];
    int age;
    char gender[2];
    char hobby[100];
};

void input_student(struct student s[], int d) {
    while (1) {
        printf("성을 입력하세요 : ");
        scanf("%s", s[d].name);

        printf("나이를 입력하세요 : ");
        scanf("%d", &s[d].age);
        if (s[d].age >= 140 || s[d].age < 0) {
            printf("살아있을까요? 다시 입력하세요.\n\n");
            continue;
        }

        printf("성별을 입력하세요 (M/F) : ");
        scanf("%s", s[d].gender);
        if ((s[d].gender[0] != 'F' && s[d].gender[0] != 'M') || s[d].gender[1] != '\0') {
            printf("성별이 이상합니다. 다시 입력하세요.\n\n");
            continue;
        }

        printf("취미를 입력하세요 : ");
        scanf("%s", s[d].hobby);
        break;
    }
}

void print_student(struct student s[], int d) {
    printf("%d. %s/%d/%s/%s\n", d + 1, s[d].name, s[d].age, s[d].gender, s[d].hobby);
}

int main() {
    struct student a[3] = {
        {"hong", 30, "M", "soccer"}, {"kim", 25, "M", "music"}, {"park", 20, "F", "movie"}
    };
    struct student students[1000];
    for (int i = 0; i < 3; i++) print_student(a, i);
    for (int j = 0; j < 1000; j++) {
        input_student(students, j);
        for (int i = 0; i < 3; i++) print_student(a, i);
        for (int k = 0; k <= j; k++) print_student(students, k + 3);
    }

    return 0;
}
