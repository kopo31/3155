#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#include <unistd.h> 

void main() {
    int num[6];
    srand(time(NULL)); 
    
    for(int i = 0; i < 6; i++) { 
        num[i] = (rand() % 45) + 1;

        for(int j = 0; j < i; j++) { 
            if(num[i] == num[j]) {
                i--;
                break;
            }
        }
    }

    for(int i = 1; i < 6; i++) {  
        int num1 = num[i];  
        int j = i - 1;

        while(j >= 0 && num[j] > num1) {
            num[j + 1] = num[j]; 
            j--;  
        }
        num[j + 1] = num1; 
    }

    for(int k = 0; k < 6; k++) {
        printf("%d\n", num[k]);  
        sleep(1);  
    }
    
    return;
}
