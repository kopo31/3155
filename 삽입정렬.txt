#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main()
{
    int value = 0;
    int number[6];
    int j = 0;
 
    for (int i = 0; i < 6; i++) {
        srand(time(NULL));
        number[i] = (int)(rand() % 45) + 1;
        for (j = 0; j < i; j++) {
            if (number[i] == number[j]) {
                i--;
            }
        }
        printf("%d ", number[i]);
        sleep(1);
    }
    printf("\n**********************\n");
    for (int i = 0; i < 5; i++) {
        value = number[i + 1];
        for (j = i; j > -1; j--) {
            if (number[j] > value) {
                number[j + 1] = number[j];
            } else {
                break;
            }
        }
        number[j + 1] = value;
    }
    
    for (int i = 0; i < 6; i++) {
        printf("%d ", number[i]);  
    } 
    return 0;
}
