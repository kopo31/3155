#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#include <unistd.h> 

void main() {
    int num[6];
    srand(time(NULL)); 
    
    for (int i = 0; i < 6; i++) { 
        num[i] = (rand() % 45) + 1;

        for (int j = 0; j < i; j++) { 
            if(num[i] == num[j]) {
                i--;
                break;
            }
        }
    }

    for (int i = 1; i < 6; i++) {  
        int num1 = num[i];  
        int j;
        for (j = i - 1; j >= 0 && num[j] > num1; j--) {  // i번째 숫자가 이전 숫자보다 작으면 자리를 바꾼다
            num[j + 1] = num[j];  // 자리 바꿈
        }
        num[j + 1] = num1; 
    }

    for (int k = 0; k < 6; k++) {
        printf("%d\n", num[k]);  
        sleep(1);  
    }
}
